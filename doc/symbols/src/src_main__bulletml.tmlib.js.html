 92         createTicker: function(config, action) {
 93             var topLabels = this._bulletml.getTopActionLabels();
 94             if (action === undefined && topLabels.length > 0) {
 95                 // top***対応.
 96                 // actionラベルtop***が定義されていた場合、それらを同時に動かす.
 97                 var tickers = [];
 98                 for ( var i = 0, end = topLabels.length; i < end; i++) {
 99                     tickers[tickers.length] = this._createTicker(config, topLabels[i]);
100                 }
101                 var parentTicker = function() {
102                     for ( var i = tickers.length; i--; ) {
103                         tickers[i].call(this);
104                     }
105                     if (parentTicker.compChildCount == tickers.length) {
106                         parentTicker.completed = true;
107                         this.dispatchEvent(tm.event.Event("completeattack"));
108                     }
109                 };
110                 for ( var i = tickers.length; i--; ) {
111                     tickers[i].parentTicker = parentTicker;
112                 }
113 
114                 parentTicker.compChildCount = 0;
115                 parentTicker.completeChild = function() {
116                     this.compChildCount++;
117                 };
118 
119                 parentTicker.compChildCount = 0;
120                 parentTicker.completed = false;
121                 parentTicker.isDanmaku = true;
122 
123                 return parentTicker;
124             } else {
125                 return this._createTicker(config, action);
126             }

128         /**
129          * @private
130          */
131         _createTicker: function(config, action) {
132             config = (function(base) {
133                 var result = {};
134                 var def = tm.bulletml.AttackPattern.defaultConfig;
135                 for ( var prop in def) {
136                     if (def.hasOwnProperty(prop)) {
137                         result[prop] = def[prop];
138                     }
139                 }
140                 for ( var prop in base) {
141                     if (base.hasOwnProperty(prop)) {
142                         result[prop] = base[prop];
143                     }
144                 }
145 
146                 return result;
147             })(config);
148 
149             if (!config.target) {
150                 throw new Error("target is undefined in config.");
151             }
152 
153             var ticker = function() {
154                 ticker.age += 1;
155                 this.age = ticker.age;
156 
157                 var conf = ticker.config;
158                 var ptn = ticker._pattern;
159 
160                 if (!ptn) {
161                     return;
162                 }
163 
164                 // update direction
165                 if (ticker.age < ticker.chDirEnd) {
166                     ticker.direction += ticker.dirIncr;
167                 } else if (ticker.age === ticker.chDirEnd) {
168                     ticker.direction = ticker.dirFin;
169                 }
170 
171                 // update speed
172                 if (ticker.age < ticker.chSpdEnd) {
173                     ticker.speed += ticker.spdIncr;
174                 } else if (ticker.age === ticker.chSpdEnd) {
175                     ticker.speed = ticker.spdFin;
176                 }
177 
178                 // update accel
179                 if (ticker.age < ticker.aclEnd) {
180                     ticker.speedH += ticker.aclIncrH;
181                     ticker.speedV += ticker.aclIncrV;
182                 } else if (ticker.age === ticker.aclEnd) {
183                     ticker.speedH = ticker.aclFinH;
184                     ticker.speedV = ticker.aclFinV;
185                 }
186 
187                 // move sprite
188                 this.x += Math.cos(ticker.direction) * ticker.speed * conf.speedRate;
189                 this.y += Math.sin(ticker.direction) * ticker.speed * conf.speedRate;
190                 this.x += ticker.speedH * conf.speedRate;
191                 this.y += ticker.speedV * conf.speedRate;
192 
193                 // test out of world
194                 if (!conf.isInsideOfWorld(this)) {
195                     this.remove();
196                     ticker.completed = true;
197                     if (ticker.parentTicker) {
198                         ticker.parentTicker.completeChild();
199                     } else {
200                         this.dispatchEvent(tm.event.Event("completeattack"));
201                     }
202                     return;
203                 }
204 
205                 // set direction, speed to bullet
206                 if (conf.updateProperties || this.updateProperties) {
207                     this.rotation = (ticker.direction + Math.PI * 0.5) * Math.RAD_TO_DEG;
208                     this.speed = ticker.speed;
209                 }
210 
211                 // proccess walker
212                 if (ticker.age < ticker.waitTo || ticker.completed) {
213                     return;
214                 }
215                 var cmd;
216                 while (cmd = ticker.walker.next()) {
217                     switch (cmd.commandName) {
218                     case "fire":
219                         ptn._fire.call(this, cmd, conf, ticker, ptn);
220                         break;
221                     case "wait":
222                         var v = 0;
223                         if (typeof(cmd.value) === "number") {
224                             ticker.waitTo = ticker.age + cmd.value;
225                         } else if ((v = ~~(cmd.value)) !== 0) {
226                             ticker.waitTo = ticker.age + v;
227                         } else {
228                             ticker.waitTo = ticker.age + eval(cmd.value);
229                         }
230                         return;
231                     case "changeDirection":
232                         ptn._changeDirection.call(this, cmd, conf, ticker);
233                         break;
234                     case "changeSpeed":
235                         ptn._changeSpeed.call(this, cmd, ticker);
236                         break;
237                     case "accel":
238                         ptn._accel.call(this, cmd, ticker);
239                         break;
240                     case "vanish":
241                         this.remove();
242                         this.dispatchEvent(tm.event.Event("removed"));
243                         break;
244                     case "notify":
245                         ptn._notify.call(this, cmd);
246                         break;
247                     }
248                 }
249 
250                 // complete
251                 ticker.completed = true;
252                 if (ticker.parentTicker) {
253                     ticker.parentTicker.completeChild();
254                 } else {
255                     this.dispatchEvent(tm.event.Event("completeattack"));
256                 }
257             };
258 
259             action = action || "top";
260             if (typeof (action) === "string") {
261                 ticker.walker = this._bulletml.getWalker(action, config.rank);
262             } else if (action instanceof bulletml.Bullet) {
263                 ticker.walker = action.getWalker(config.rank);
264             } else {
265                 window.console.error(config, action);
266                 throw new Error("引数が不正");
267             }
268 
269             ticker._pattern = this;
270             ticker.config = config;
271             ticker.waitTo = -1;
272             ticker.completed = false;
273             ticker.direction = 0;
274             ticker.lastDirection = 0;
275             ticker.speed = 0;
276             ticker.lastSpeed = 0;
277             ticker.speedH = 0;
278             ticker.speedV = 0;
279             ticker.dirIncr = 0;
280             ticker.dirFin = 0;
281             ticker.chDirEnd = -1;
282             ticker.spdIncr = 0;
283             ticker.spdFin = 0;
284             ticker.chSpdEnd = -1;
285             ticker.aclIncrH = 0;
286             ticker.aclFinH = 0;
287             ticker.aclIncrV = 0;
288             ticker.aclFinV = 0;
289             ticker.aclEnd = -1;
290             ticker.age = -1;
291 
292             ticker.isDanmaku = true;
293             return ticker;
294         },

342         /**
343          * @private
344          */
345         _fire: function(cmd, config, ticker, pattern) {
346             var spec = { label: cmd.bullet.label };
347             for (var key in cmd.bullet.option) {
348                 spec[key] = cmd.bullet.option[key];
349             }
350 
351             var b = config.bulletFactory(spec);
352             if (!b) {
353                 return;
354             }
355 
356             // 等速直進弾?
357             var uniformLinearBullet = !!cmd.bullet.actions.length;
358 
359             var bt = uniformLinearBullet ? (
360                 pattern._createSimpleTicker(config)
361             ) : (
362                 pattern.createTicker(config, cmd.bullet)
363             );
364 
365             var attcker = this;
366             var gunPosition = {
367                 x: this.x + cmd.option.offsetX,
368                 y: this.y + cmd.option.offsetY
369             };
370 
371             var calcDirection = function(d) {
372                 var dv = eval(d.value) * Math.DEG_TO_RAD;
373                 // console.debug(d.type);
374                 switch(d.type) {
375                 case "aim":
376                     if (config.target) {
377                         if (cmd.option.autonomy) {
378                             return angleAtoB(gunPosition, config.target) + dv;
379                         } else {
380                             return angleAtoB(attcker, config.target) + dv;
381                         }
382                     } else {
383                         return dv - Math.PI / 2;
384                     }
385                 case "absolute":
386                     return dv - Math.PI / 2; // 真上が0度
387                 case "relative":
388                     return ticker.direction + dv;
389                 case "sequence":
390                 default:
391                     // console.debug(ticker.lastDirection, dv);
392                     return ticker.lastDirection + dv;
393                 }
394             };
395             ticker.lastDirection = bt.direction = calcDirection(cmd.direction || cmd.bullet.direction);
396             // console.debug(bt.direction);
397 
398             var calcSpeed = function(s) {
399                 var sv = eval(s.value);
400                 switch (s.type) {
401                 case "relative":
402                     return ticker.speed + sv;
403                 case "sequence":
404                     return ticker.lastSpeed + sv;
405                 case "absolute":
406                 default:
407                     return sv;
408                 }
409             };
410             ticker.lastSpeed = bt.speed = calcSpeed(cmd.speed || cmd.bullet.speed);
411 
412             b.x = gunPosition.x;
413             b.y = gunPosition.y;
414 
415             if (uniformLinearBullet) {
416                 bt.deltaX = Math.cos(bt.direction) * bt.speed * config.speedRate;
417                 bt.deltaY = Math.sin(bt.direction) * bt.speed * config.speedRate;
418             }
419 
420             // set direction, speed to bullet
421             b.updateProperties = !!b.updateProperties;
422             if (config.updateProperties || b.updateProperties) {
423                 b.rotation = (bt.direction + Math.PI * 0.5) * Math.RAD_TO_DEG;
424                 b.speed = bt.speed;
425             }
426 
427             b.addEventListener("enterframe", bt);
428             b.addEventListener("removed", function() {
429                 this.removeEventListener("enterframe", bt);
430                 this.removeEventListener("removed", arguments.callee);
431             });
432             if (config.addTarget) {
433                 config.addTarget.addChild(b);
434             } else if (this.parent) {
435                 this.parent.addChild(b);
436             }
437         },
438         /**
439          * @private
440          */
441         _changeDirection: function(cmd, config, ticker) {
442             var d = eval(cmd.direction.value) * Math.DEG_TO_RAD;
443             var t = eval(cmd.term);
444             switch (cmd.direction.type) {
445             case "aim":
446                 var tar = config.target;
447                 if (!tar) {
448                     return;
449                 }
450                 ticker.dirFin = angleAtoB(this, tar) + d;
451                 ticker.dirIncr = normalizeRadian(ticker.dirFin - ticker.direction) / t;
452                 break;
453             case "absolute":
454                 ticker.dirFin = d - Math.PI / 2;
455                 ticker.dirIncr = normalizeRadian(ticker.dirFin - ticker.direction) / t;
456                 break;
457             case "relative":
458                 ticker.dirFin = ticker.direction + d;
459                 ticker.dirIncr = normalizeRadian(ticker.dirFin - ticker.direction) / t;
460                 break;
461             case "sequence":
462                 ticker.dirIncr = d;
463                 ticker.dirFin = ticker.direction + ticker.dirIncr * (t-1);
464                 break;
465             }
466             ticker.chDirEnd = ticker.age + t;
467         },
468         /**
469          * @private
470          */
471         _changeSpeed: function(cmd, ticker) {
472             // console.log("changeSpeed")
473             var s = eval(cmd.speed.value);
474             var t = eval(cmd.term);
475             switch (cmd.speed.type) {
476             case "absolute":
477                 ticker.spdFin = s;
478                 ticker.spdIncr = (ticker.spdFin - ticker.speed) / t;
479                 break;
480             case "relative":
481                 ticker.spdFin = s + ticker.speed;
482                 ticker.spdIncr = (ticker.spdFin - ticker.speed) / t;
483                 break;
484             case "sequence":
485                 ticker.spdIncr = s;
486                 ticker.spdFin = ticker.speed + ticker.spdIncr * t;
487                 break;
488             }
489             ticker.chSpdEnd = ticker.age + t;
490         },
491         /**
492          * @private
493          */
494         _accel: function(cmd, ticker) {
495             var t = eval(cmd.term);
496             ticker.aclEnd = ticker.age + t;
497 
498             if (cmd.horizontal) {
499                 var h = eval(cmd.horizontal.value);
500                 switch (cmd.horizontal.type) {
501                 case "absolute":
502                 case "sequence":
503                     ticker.aclIncrH = (h - ticker.speedH) / t;
504                     ticker.aclFinH = h;
505                     break;
506                 case "relative":
507                     ticker.aclIncrH = h;
508                     ticker.aclFinH = (h - ticker.speedH) * t;
509                     break;
510                 }
511             } else {
512                 ticker.aclIncrH = 0;
513                 ticker.aclFinH = ticker.speedH;
514             }
515 
516             if (cmd.vertical) {
517                 var v = eval(cmd.vertical.value);
518                 switch (cmd.vertical.type) {
519                 case "absolute":
520                 case "sequence":
521                     ticker.aclIncrV = (v - ticker.speedV) / t;
522                     ticker.aclFinV = v;
523                     break;
524                 case "relative":
525                     ticker.aclIncrV = v;
526                     ticker.aclFinV = (v - ticker.speedV) * t;
527                     break;
528                 }
529             } else {
530                 ticker.aclIncrV = 0;
531                 ticker.aclFinV = ticker.speedV;
532             }
533         },
534         /**
535          * @private
536          */
537         _notify: function(cmd) {
538             var e = tm.event.Event(cmd.eventName);
539             if (cmd.params) {
540                 for (var key in cmd.params) {
541                     e[key] = cmd.params[key];
542                 }
543             }
544             this.dispatchEvent(e);
545         }

592     /**
593      * ラジアンを -π<= rad < π の範囲に正規化する.
594      */
595     function normalizeRadian(radian) {
596         while (radian <= -Math.PI) {
597             radian += Math.PI * 2;
598         }
599         while (Math.PI < radian) {
600             radian -= Math.PI * 2;
601         }
602         return radian;
603     }

605     /**
606      * スプライトAから見たスプライトBの方向をラジアンで返す.
607      *
608      * @param {tm.app.CanvasElement}
609      *            a スプライトA
610      * @param {tm.app.CanvasElement}
611      *            b スプライトB
612      */
613     function angleAtoB(a, b) {
614         return Math.atan2(b.y-a.y, b.x-a.x);
615     }
