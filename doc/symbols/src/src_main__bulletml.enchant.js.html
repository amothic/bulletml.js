  8     // BulletML(*.bml, *.xml)をpreloadで読み込めるようにする.
  9     enchant.Game._loadFuncs["bml"] = enchant.Game._loadFuncs["xml"] = function(
 10             src, callback) {
 11         var game = this;
 12         var xhr = new XMLHttpRequest();
 13         xhr.onreadystatechange = function(e) {
 14             if (xhr.readyState === 4) {
 15                 if (xhr.status !== 200 && xhr.status !== 0) {
 16                     throw new Error(xhr.status + ': '
 17                             + 'Cannot load an asset: ' + src);
 18                 }
 19 
 20                 if (xhr.responseXML != null) {
 21                     var root = bulletml.build(xhr.responseXML);
 22                     if (root) {
 23                         game.assets[src] = new enchant.bulletml.AttackPattern(root);
 24                     } else {
 25                         alert(src + "は妥当なBulletMLではありません。");
 26                         game.assets[src] = xhr.responseXML;
 27                     }
 28                     callback();
 29                 } else {
 30                     throw new Error(xhr.status + ': '
 31                             + 'Cannot load an asset: ' + src);
 32                 }
 33             }
 34         };
 35 
 36         xhr.open('GET', src, true);
 37         if (xhr.overrideMimeType) {
 38             xhr.overrideMimeType("application/xml");
 39         }
 40         xhr.send(null);
 41     };

 55         if (!this._listeners["enterframe"]
 56                 || this._listeners["enterframe"].length === 0) {
 57             return;
 58         }
 59         var remove = [];
 60         for ( var i = this._listeners["enterframe"].length; i--;) {
 61             if (this._listeners["enterframe"][i].isDanmaku) {
 62                 remove[remove.length] = this._listeners["enterframe"][i];

178         createTicker : function(config, action) {
179             if (config instanceof enchant.Node) {
180                 config = {
181                     target: config
182                 };
183             }
184             var topLabels = this._bulletml.getTopActionLabels()
185             if (action === undefined && topLabels.length > 1) {
186                 // top***対応.
187                 // actionラベルtop***が定義されていた場合、それらを同時に動かす.
188                 var tickers = [];
189                 for ( var i = 0, end = topLabels.length; i < end; i++) {
190                     tickers[tickers.length] = this._createTicker(config, topLabels[i]);
191                 }
192                 var parentTicker = function() {
193                     for ( var i = tickers.length; i--;) {
194                         tickers[i].call(this);
195                     }
196                     if (parentTicker.compChildCount === tickers.length) {
197                         parentTicker.complete = true;
198                         this.dispatchEvent(new Event("completeattack"));
199                     }
200                 };
201                 for ( var i = tickers.length; i--;) {
202                     tickers[i].parentTicker = parentTicker;
203                 }
204 
205                 parentTicker.compChildCount = 0;
206                 parentTicker.completeChild = function() {
207                     this.compChildCount++;
208                 };
209 
210                 parentTicker.compChildCount = 0;
211                 parentTicker.complete = false;
212                 parentTicker.isDanmaku = true;
213 
214                 return parentTicker;
215             } else {
216                 return this._createTicker(config, action);
217             }

222         _createTicker : function(config, action) {
223             config = (function(base) {
224                 var result = {};
225                 var def = enchant.bulletml.AttackPattern.defaultConfig;
226                 for ( var prop in def) {
227                     if (def.hasOwnProperty(prop)) {
228                         result[prop] = def[prop];
229                     }
230                 }
231                 for ( var prop in base) {
232                     if (base.hasOwnProperty(prop)) {
233                         result[prop] = base[prop];
234                     }
235                 }
236 
237                 return result;
238             })(config);
239 
240             if (!config.target) {
241                 throw new Error("target is undefined in config.");
242             }
243 
244             var ticker = function() {
245                 var conf = ticker.config;
246                 var ptn = ticker._pattern;
247 
248                 if (!ptn) {
249                     return;
250                 }
251 
252                 // update direction
253                 if (this.age < ticker.chDirEnd) {
254                     ticker.direction += ticker.dirIncr;
255                 } else if (this.age === ticker.chDirEnd) {
256                     ticker.direction = ticker.dirFin;
257                 }
258 
259                 // update speed
260                 if (this.age < ticker.chSpdEnd) {
261                     ticker.speed += ticker.spdIncr;
262                 } else if (this.age === ticker.chSpdEnd) {
263                     ticker.speed = ticker.spdFin;
264                 }
265 
266                 // update accel
267                 if (this.age < ticker.aclEnd) {
268                     ticker.speedH += ticker.aclIncrH;
269                     ticker.speedV += ticker.aclIncrV;
270                 } else if (this.age === ticker.aclEnd) {
271                     ticker.speedH = ticker.aclFinH;
272                     ticker.speedV = ticker.aclFinV;
273                 }
274 
275                 // move sprite
276                 this.x += Math.cos(ticker.direction) * ticker.speed * conf.speedRate;
277                 this.y += Math.sin(ticker.direction) * ticker.speed * conf.speedRate;
278                 this.x += ticker.speedH * conf.speedRate;
279                 this.y += ticker.speedV * conf.speedRate;
280 
281                 // test out of world
282                 if (!conf.isInsideOfWorld(this)) {
283                     if (this.parentNode) this.parentNode.removeChild(this);
284                     ticker.completed = true;
285                     if (ticker.parentTicker) {
286                         ticker.parentTicker.completeChild();
287                     } else {
288                         this.dispatchEvent(new Event("completeattack"));
289                     }
290                     return;
291                 }
292 
293                 // set direction, speed to bullet
294                 if (conf.updateProperties || this.updateProperties) {
295                     this.rotation = (ticker.direction) * RAD_TO_DEG;
296                     this.speed = ticker.speed;
297                 }
298 
299                 // proccess walker
300                 if (this.age < ticker.waitTo || ticker.completed) {
301                     return;
302                 }
303                 var cmd;
304                 while (cmd = ticker.walker.next()) {
305                     switch (cmd.commandName) {
306                     case "fire":
307                         ptn._fire.call(this, cmd, conf, ticker, ptn);
308                         break;
309                     case "wait":
310                         var v = 0;
311                         if (typeof(cmd.value) === 'number') {
312                             ticker.waitTo = this.age + cmd.value;
313                         } else if ((v = ~~(cmd.value)) !== 0) {
314                             ticker.waitTo = this.age + v;
315                         } else {
316                             ticker.waitTo = this.age + eval(cmd.value);
317                         }
318                         return;
319                     case "changeDirection":
320                         ptn._changeDirection.call(this, cmd, conf, ticker);
321                         break;
322                     case "changeSpeed":
323                         ptn._changeSpeed.call(this, cmd, ticker);
324                         break;
325                     case "accel":
326                         ptn._accel.call(this, cmd, ticker);
327                         break;
328                     case "vanish":
329                         if (this.parentNode) this.parentNode.removeChild(this);
330                         break;
331                     }
332                 }
333 
334                 // complete
335                 ticker.completed = true;
336                 if (ticker.parentTicker) {
337                     ticker.parentTicker.completeChild();
338                 } else {
339                     this.dispatchEvent(new Event("completeattack"));
340                 }
341             };
342 
343             action = action || "top";
344             if (typeof (action) === "string") {
345                 ticker.walker = this._bulletml.getWalker(action, config.rank);
346             } else if (action instanceof bulletml.Bullet) {
347                 ticker.walker = action.getWalker(config.rank);
348             } else {
349                 window.console.error(config, action);
350                 throw new Error("引数が不正");
351             }
352 
353             ticker._pattern = this;
354             ticker.config = config;
355             ticker.waitTo = -1;
356             ticker.completed = false;
357             ticker.direction = 0;
358             ticker.lastDirection = 0;
359             ticker.speed = 0;
360             ticker.lastSpeed = 0;
361             ticker.speedH = 0;
362             ticker.speedV = 0;
363             ticker.dirIncr = 0;
364             ticker.dirFin = 0;
365             ticker.chDirEnd = -1;
366             ticker.spdIncr = 0;
367             ticker.spdFin = 0;
368             ticker.chSpdEnd = -1;
369             ticker.aclIncrH = 0;
370             ticker.aclFinH = 0;
371             ticker.aclIncrV = 0;
372             ticker.aclFinV = 0;
373             ticker.aclEnd = -1;
374 
375             ticker.isDanmaku = true;
376             return ticker;
377         },

428         _fire : function(cmd, config, ticker, pattern) {
429             var spec = { label: cmd.bullet.label };
430             for (var key in cmd.bullet.option) {
431                 spec[key] = cmd.bullet.option[key];
432             }
433 
434             var b = config.bulletFactory(spec);
435             if (!b) {
436                 return;
437             }
438 
439             // 等速直進弾?
440             var uniformLinearBullet = !!cmd.bullet.actions.length;
441 
442             var bt = uniformLinearBullet ? (
443                 pattern._createSimpleTicker(config)
444             ) : (
445                 pattern.createTicker(config, cmd.bullet)
446             );
447 
448             var attacker = this;
449             var calcDirection = function(d) {
450                 var dv = eval(d.value) * DEG_TO_RAD;
451                 // console.debug(d.type);
452                 switch (d.type) {
453                 case "aim":
454                     if (config.target) {
455                         return angleAtoB(attacker, config.target) + dv;
456                     } else {
457                         return dv - Math.PI / 2;
458                     }
459                 case "absolute":
460                     return dv - Math.PI / 2; // 真上が0度
461                 case "relative":
462                     return ticker.direction + dv;
463                 case "sequence":
464                 default:
465                     // console.debug(ticker.lastDirection, dv);
466                     return ticker.lastDirection + dv;
467                 }
468             };
469             ticker.lastDirection = bt.direction = calcDirection(cmd.direction || cmd.bullet.direction);
470             // console.debug(bt.direction);
471 
472             var calcSpeed = function(s) {
473                 var sv = eval(s.value);
474                 switch (s.type) {
475                 case "relative":
476                     return ticker.speed + sv;
477                 case "sequence":
478                     return ticker.lastSpeed + sv;
479                 case "absolute":
480                 default:
481                     return sv;
482                 }
483             };
484             ticker.lastSpeed = bt.speed = calcSpeed(cmd.speed || cmd.bullet.speed);
485 
486             b.x = this.x + ((this.width || 0) - (b.width || 0)) / 2;
487             b.y = this.y + ((this.height || 0) - (b.height || 0)) / 2;
488 
489             if (uniformLinearBullet) {
490                 bt.deltaX = Math.cos(bt.direction) * bt.speed * config.speedRate;
491                 bt.deltaY = Math.sin(bt.direction) * bt.speed * config.speedRate;
492             }
493 
494             // set direction, speed to bullet
495             this.updateProperties = !!this.updateProperties;
496             if (config.updateProperties || this.updateProperties) {
497                 b.rotation = (ticker.direction) * RAD_TO_DEG;
498                 b.speed = ticker.speed;
499             }
500 
501             b.addEventListener("enterframe", bt);
502             b.addEventListener("removed", function() {
503                 this.removeEventListener("enterframe", bt);
504             });
505             if (config.addTarget) {
506                 config.addTarget.addChild(b);
507             } else if (this.parentNode) {
508                 this.parentNode.addChild(b);
509             }

511         /**

513          */
514         _changeDirection : function(cmd, config, ticker) {
515             var d = eval(cmd.direction.value) * DEG_TO_RAD;
516             var t = eval(cmd.term);
517             switch (cmd.direction.type) {
518             case "aim":
519                 var tar = config.target;
520                 if (!tar) {
521                     return;
522                 }
523                 ticker.dirFin = angleAtoB(this, tar) + d;
524                 ticker.dirIncr = normalizeRadian(ticker.dirFin - ticker.direction) / t;
525                 break;
526             case "absolute":
527                 ticker.dirFin = d - Math.PI / 2;
528                 ticker.dirIncr = normalizeRadian(ticker.dirFin - ticker.direction) / t;
529                 break;
530             case "relative":
531                 ticker.dirFin = ticker.direction + d;
532                 ticker.dirIncr = normalizeRadian(ticker.dirFin - ticker.direction) / t;
533                 break;
534             case "sequence":
535                 ticker.dirIncr = d;
536                 ticker.dirFin = ticker.direction + ticker.dirIncr * (t-1);
537                 break;
538             }
539             ticker.chDirEnd = this.age + t;
540         },
541         /**
542          * @private
543          */
544         _changeSpeed : function(cmd, ticker) {
545             var s = eval(cmd.speed.value);
546             var t = eval(cmd.term);
547             switch (cmd.speed.type) {
548             case "absolute":
549                 ticker.spdFin = s;
550                 ticker.spdIncr = (ticker.spdFin - ticker.speed) / t;
551                 break;
552             case "relative":
553                 ticker.spdFin = s + ticker.speed;
554                 ticker.spdIncr = (ticker.spdFin - ticker.speed) / t;
555                 break;
556             case "sequence":
557                 ticker.spdIncr = s;
558                 ticker.spdFin = ticker.speed + ticker.spdIncr * t;
559                 break;
560             }
561             ticker.chSpdEnd = this.age + t;
562         },
563         /**
564          * @private
565          */
566         _accel : function(cmd, ticker) {
567             var t = eval(cmd.term);
568             ticker.aclEnd = this.age + t;
569 
570             if (cmd.horizontal) {
571                 var h = eval(cmd.horizontal.value);
572                 switch (cmd.horizontal.type) {
573                 case "absolute":
574                 case "sequence":
575                     ticker.aclIncrH = (h - ticker.speedH) / t;
576                     ticker.aclFinH = h;
577                     break;
578                 case "relative":
579                     ticker.aclIncrH = h;
580                     ticker.aclFinH = (h - ticker.speedH) * t;
581                     break;
582                 }
583             } else {
584                 ticker.aclIncrH = 0;
585                 ticker.aclFinH = ticker.speedH;
586             }
587 
588             if (cmd.vertical) {
589                 var v = eval(cmd.vertical.value);
590                 switch (cmd.vertical.type) {
591                 case "absolute":
592                 case "sequence":
593                     ticker.aclIncrV = (v - ticker.speedV) / t;
594                     ticker.aclFinV = v;
595                     break;
596                 case "relative":
597                     ticker.aclIncrV = v;
598                     ticker.aclFinV = (v - ticker.speedV) * t;
599                     break;
600                 }
601             } else {
602                 ticker.aclIncrV = 0;
603                 ticker.aclFinV = ticker.speedV;
604             }

658     var RAD_TO_DEG = 180 / Math.PI;

661     /**
662      * ラジアンを -π<= rad < π の範囲に正規化する.
663      */
664     function normalizeRadian(radian) {
665         while (radian <= -Math.PI) {
666             radian += Math.PI * 2;
667         }
668         while (Math.PI < radian) {
669             radian -= Math.PI * 2;
670         }
671         return radian;
672     }
673     /**
674      * スプライトAから見たスプライトBの方向をラジアンで返す.
675      *
676      * @param {enchant.Node}
677      *            a スプライトA
678      * @param {enchant.Node}
679      *            b スプライトB
680      */
681     function angleAtoB(a, b) {
682         var ca = {
683             x : a.x + (a.width || 0) / 2,
684             y : a.y + (a.height || 0) / 2
685         };
686         var cb = {
687             x : b.x + (b.width || 0) / 2,
688             y : b.y + (b.height || 0) / 2
689         };
690         return Math.atan2(cb.y - ca.y, cb.x - ca.x);
